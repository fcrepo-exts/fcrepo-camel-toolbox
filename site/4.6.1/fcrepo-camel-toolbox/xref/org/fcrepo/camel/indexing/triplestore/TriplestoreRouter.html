18  package org.fcrepo.camel.indexing.triplestore;
19  
20  import static org.apache.camel.builder.PredicateBuilder.not;
21  import static org.apache.camel.builder.PredicateBuilder.or;
22  import static org.fcrepo.camel.FcrepoHeaders.FCREPO_NAMED_GRAPH;
23  import static org.fcrepo.camel.JmsHeaders.EVENT_TYPE;
24  import static org.fcrepo.camel.JmsHeaders.IDENTIFIER;
25  import static org.fcrepo.camel.RdfNamespaces.INDEXING;
26  import static org.fcrepo.camel.RdfNamespaces.RDF;
27  import static org.fcrepo.camel.RdfNamespaces.REPOSITORY;
28  import static org.slf4j.LoggerFactory.getLogger;
29  
30  import org.apache.camel.LoggingLevel;
31  import org.apache.camel.builder.RouteBuilder;
32  import org.apache.camel.builder.xml.Namespaces;
33  import org.apache.camel.builder.xml.XPathBuilder;
34  import org.fcrepo.camel.processor.SparqlDeleteProcessor;
35  import org.fcrepo.camel.processor.SparqlUpdateProcessor;
36  import org.slf4j.Logger;
37  
38  /**
39   * A content router for handling JMS events.
40   *
41   * @author Aaron Coburn
42   */
43  public class TriplestoreRouter extends RouteBuilder {
44  
45      private static final Logger logger = getLogger(TriplestoreRouter.class);
46  
47      private static final String RESOURCE_DELETION = "http://fedora.info/definitions/v4/event#ResourceDeletion";
48  
49      /**
50       * Configure the message route workflow.
51       */
52      public void configure() throws Exception {
53  
54          final Namespaces ns = new Namespaces("rdf", RDF);
55          ns.add("indexing", INDEXING);
56  
57          final XPathBuilder indexable = new XPathBuilder(
58                  String.format("/rdf:RDF/rdf:Description/rdf:type[@rdf:resource='%s']", INDEXING + "Indexable"));
59          indexable.namespaces(ns);
60  
61          /**
62           * A generic error handler (specific to this RouteBuilder)
63           */
64          onException(Exception.class)
65              .maximumRedeliveries("{{error.maxRedeliveries}}")
66              .log("Index Routing Error: ${routeId}");
67  
68          /**
69           * route a message to the proper queue, based on whether
70           * it is a DELETE or UPDATE operation.
71           */
72          from("{{input.stream}}")
73              .routeId("FcrepoTriplestoreRouter")
74              .choice()
75                  // this clause supports Fedora 4.5.1 and earlier but may be removed in a future release
76                  .when(header(EVENT_TYPE).isEqualTo(REPOSITORY + "NODE_REMOVED"))
77                      .to("direct:delete.triplestore")
78                  .when(header(EVENT_TYPE).isEqualTo(RESOURCE_DELETION))
79                      .to("direct:delete.triplestore")
80                  .otherwise()
81                      .to("direct:index.triplestore");
82  
83          /**
84           * Handle re-index events
85           */
86          from("{{triplestore.reindex.stream}}")
87              .routeId("FcrepoTriplestoreReindex")
88              .to("direct:index.triplestore");
89  
90          /**
91           * Based on an item's metadata, determine if it is indexable.
92           */
93          from("direct:index.triplestore")
94              .routeId("FcrepoTriplestoreIndexer")
95              .filter(not(or(header(IDENTIFIER).startsWith(simple("{{audit.container}}/")),
96                      header(IDENTIFIER).isEqualTo(simple("{{audit.container}}")))))
97              .removeHeaders("CamelHttp*")
98              .to("fcrepo:{{fcrepo.baseUrl}}?preferInclude=PreferMinimalContainer&accept=application/rdf+xml")
99              .choice()
100                 .when(or(simple("{{indexing.predicate}} != 'true'"), indexable))
101                     .to("direct:update.triplestore")
102                 .otherwise()
103                     .to("direct:delete.triplestore");
104 
105         /**
106          * Remove an item from the triplestore index.
107          */
108         from("direct:delete.triplestore")
109             .routeId("FcrepoTriplestoreDeleter")
110             .process(new SparqlDeleteProcessor())
111             .log(LoggingLevel.INFO, logger,
112                     "Deleting Triplestore Object ${headers[CamelFcrepoIdentifier]} " +
113                     "${headers[org.fcrepo.jms.identifier]}")
114             .to("{{triplestore.baseUrl}}?useSystemProperties=true");
115 
116         /**
117          * Perform the sparql update.
118          */
119         from("direct:update.triplestore")
120             .routeId("FcrepoTriplestoreUpdater")
121             .setHeader(FCREPO_NAMED_GRAPH)
122                 .simple("{{triplestore.namedGraph}}")
123             .to("fcrepo:{{fcrepo.baseUrl}}?accept=application/n-triples" +
124                     "&preferOmit={{prefer.omit}}&preferInclude={{prefer.include}}")
125             .process(new SparqlUpdateProcessor())
126             .log(LoggingLevel.INFO, logger,
127                     "Indexing Triplestore Object ${headers[CamelFcrepoIdentifier]} " +
128                     "${headers[org.fcrepo.jms.identifier]}")
129             .to("{{triplestore.baseUrl}}?useSystemProperties=true");
130     }
131 }
